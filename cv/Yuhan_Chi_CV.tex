\section*{Selected Open-Source Projects}

\project{\href{https://github.com/Chi-Shan0707/token-verification-mirage}{token-verification-mirage}}{\href{https://openreview.net/pdf?id=wRImV3kfR1}{Paper} $\cdot$ \href{https://ai4math2026.github.io/}{AI4Math Workshop}}
\begin{itemize}
  \item Poster, \textbf{ICML 2026 Workshop on AI for Math (AI4Math)}; solo-authored, full pipeline (dataset selection, generation, evaluation design, analysis, writing)
  \item Evaluation-protocol choices alone (pooling, in-sample scoring, direction-agnostic AUROC) shift apparent verification AUROC by up to \textbf{0.18}

\project{\href{https://github.com/chi-shan0707/code-not-text}{code-not-text}}{\href{https://chi-shan0707.github.io/code-not-text/demo}{Demo}}
\begin{itemize}
  \item Cross-domain measurement study across 31,040 math, science, and coding runs: AoA 0.958 (math), 0.799 (science), 0.434 (coding, below the random-direction baseline)
  \item Rules out five alternative explanations for the coding failure (model capacity, label scarcity, surface noise, feature coverage, judge framing) --- all converge to the same ceiling; frames the result as measurement non-invariance
\end{itemize}
