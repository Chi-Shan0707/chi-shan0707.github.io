\section*{Selected Research and Open-Source Projects}

\project{token-verification-mirage}{\href{https://openreview.net/pdf?id=wRImV3kfR1}{Paper} \;|\; \href{https://github.com/Chi-Shan0707/token-verification-mirage}{GitHub} \;|\; \href{https://ai4math2026.github.io/}{AI4Math Workshop}}
\begin{itemize}
  \item Solo-author project with full pipeline ownership: dataset selection, model deployment, trace generation, evaluation design, analysis, figures, related-work organization, and writing.
  \item Poster at the \textbf{ICML 2026 Workshop on AI for Math (AI4Math)}.
  \item Audits how evaluation protocol choices such as pooling, in-sample scoring, and direction-agnostic AUROC can change apparent token-level verification performance.
  \item Finds that shallow token statistics can be useful diagnostics, but should not be treated as stable standalone verifiers without within-problem evaluation, fixed-direction baselines, and permutation-null calibration.
\end{itemize}

\project{code-not-text}{\href{https://github.com/Chi-Shan0707/code-not-text}{GitHub} \;|\; \href{https://chi-shan0707.github.io/code-not-text/demo/}{Interactive Demo}}
\begin{itemize}
  \item Solo measurement study of cheap CoT-surface and token-trajectory features on \textbf{DeepSeek-R1-0528-Qwen3-8B} across \textbf{31,040} math, science, and coding runs.
  \item Reports strong math signal (AoA \textbf{0.958}, AUROC@100\% \textbf{0.982}), partial science signal (AoA \textbf{0.799}), and weak coding transfer (AoA \textbf{0.434}).
  \item Tests coding-side robustness through an 83-scalar feature sweep, grouped ablations, a CoT-only judge, nonlinear MLPs, SSL pre-training, semantic-knot annotation, and token-level de-knotting.
  \item Frames the result as \textbf{measurement non-invariance}: the same feature family tracks convergence-like behavior in math but not executable correctness in coding.
\end{itemize}

\project{TinyLoRA-GRPO-Coder}{\href{https://github.com/Chi-Shan0707/TinyLoRA-GRPO-Coder}{GitHub}}
\begin{itemize}
  \item Solo project adapting a small-parameter training idea toward verifiable competitive-programming code generation on Qwen2.5-Coder-3B.
  \item Uses compile-and-run rewards rather than static heuristics; built as a full research-system exercise covering data processing, training, multi-GPU setup, reward design, evaluation, and validation.
\end{itemize}
